15 float check_over_span_float(float f1(float),
16  float f2(float),
17  const float low_limit,
18  const float span,
19  const size_t npoints) {
20  float x = 0.0;
21  float val = f1(x) - f2(x);
22  float err = val * val; // 0.0 is an edge case
23 
24 
25  for (size_t i = 0; i < npoints; ++i) {
26  x = low_limit + span * i / npoints;
27  val = f1(x) - f2(x);
28  err += val * val;
29  }
30  err /= npoints + 1;
31  return err;
32 }
